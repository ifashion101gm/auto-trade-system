def calculate_vwap(ohlcv: list) -> float: """Volume-Weighted Average Price — gold intraday anchor.""" cumulative_tp_vol = sum((c[2]+c[3]+c[4])/3 * c[5] for c in ohlcv) total_vol = sum(c[5] for c in ohlcv) return cumulative_tp_vol / total_vol if total_vol > 0 else 0.0 def calculate_ema(prices: list, period: int) -> float: """Exponential Moving Average.""" k = 2 / (period + 1) ema = prices[0] for p in prices[1:]: ema = p * k + ema * (1 - k) return ema def calculate_dxy_bias(dxy_change_pct: float) -> str: """Gold/USD inverse correlation filter.
